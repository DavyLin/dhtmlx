window.onload = function() {
    var chart = new dhtmlXChart({
        view: "pie3D",
        container: "chart",
        value: "#sales#",
        label: function(obj) {
            var sum = chart.sum("#sales#");
            return obj.component + " (" + Math.round(parseFloat(obj.sales) / sum * 100) + "%)";
        },
        color: "#color#",
        radius: 110,
        tooltip: "#sales#";
    });
    chart.parse(data, "json");
}
