<script>
var data = new dhtmlXDataStore({
    url: "../common/data/data.json",
    ready: function() {
        this.setCursor(this.first());
    },
    datatype: "json";
});
function show_next() {
    data.setCursor(data.next(data.getCursor()));
}
data1 = new dhtmlXDataView({
    container: "gridbox2",
    type: {
        template: "<span class='dhx_strong'>#Maintainer#</span>#Package# <span class='dhx_light'>#Version#</span>",
        height: 35;
    }
});
 
data1.bind(data, function(master, filter) {
    return filter.Version == master.Version;
});
data1.load("../common/data/data.json", "json");
</script>
