<script>
var data = new dhtmlXDataStore({
    url: "../common/data/data.json",
    datatype: "json";
});
mygrid = new dhtmlXGridObject('gridbox');
mygrid.setSkin("dhx_skyblue");
mygrid.setHeader("Count,Name");
mygrid.setColumnIds("Maintainer,Version");
mygrid.init();
mygrid.sync(data);
var formData = [{
    type: "label",
    label: "Selected record"
}, {
    type: "input",
    name: "Maintainer",
    value: "",
    labelWidth: 100,
    label: "Maintainer"
}, {
    type: "input",
    name: "Version",
    value: "",
    labelWidth: 100,
    label: "Version"
}];
myform = new dhtmlXForm("gridbox3", formData);
myform.bind(mygrid);
 
mygrid2 = new dhtmlXGridObject('gridbox2');
mygrid2.setSkin("dhx_skyblue");
mygrid2.setHeader("Count,Name");
mygrid2.init();
mygrid2.setColumnIds("Maintainer,Package");
mygrid2.sync(data);
var formData2 = [{
    type: "label",
    label: "Selected record"
}, {
    type: "input",
    name: "Maintainer",
    value: "",
    labelWidth: 100,
    label: "Maintainer"
}, {
    type: "input",
    name: "Package",
    value: "",
    labelWidth: 100,
    label: "Package"
}];
myform2 = new dhtmlXForm("gridbox4", formData2);
myform2.bind(mygrid2);
</script>
